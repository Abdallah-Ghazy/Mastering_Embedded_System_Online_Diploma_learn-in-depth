#include <stdio.h>
int main(){
	float a = 0;
	float b = 0;
	float x = 0;

	printf("##########Console-output###\n");

	printf("Enter value of a:");
	fflush(stdout);
	scanf("%f",&a);

	printf("Enter value of b:");
	fflush(stdout);
	scanf("%f",&b);

	x = a;
	a = b;
	b = x;

	printf("After swapping, value of a = %.2f\n",a);
	printf("After swapping, value of b = %.2f\n",b);

	printf("###########################");

	return 0; }

 #include <stdio.h>
int main() {
		char character ;
		printf("Enter an alphabet:");
		fflush(stdout);
		scanf("%c",&character);

		switch(character){
		case 'a':
		case 'e':
		case 'i':
		case 'o':
		case 'u':
		case 'A':
		case 'E':
		case 'I':
		case 'O':
		case 'U':
			printf("%c  is a vowel.",character);
			break;

		default:
			printf("%c  is a consonant.",character);
		}
    return 0;
}
